Enhanced Service Architecture

// Generated MCP server with enhanced v2.0 architecture
import { 
  Logger, 
  ErrorHandler, 
  ConfigurationManager,
  HealthChecker,
  TelemetryService,
  LLMOrchestrator,
  ProjectAnalyzer,
  CodeGenerator 
} from './services';

// Modular service initialization
const logger = new Logger();
const errorHandler = new ErrorHandler(logger);
const configManager = new ConfigurationManager(logger);
const healthChecker = new HealthChecker(logger);
const telemetry = new TelemetryService(logger);
const llmOrchestrator = new LLMOrchestrator(logger, configManager);
const projectAnalyzer = new ProjectAnalyzer(logger, llmOrchestrator);
const codeGenerator = new CodeGenerator(logger, llmOrchestrator);

// AI-powered project insights
const analysis = await projectAnalyzer.analyzeProject({
  path: './my-api',
  includeAIInsights: true,
  generateRecommendations: true
});

// Enhanced health monitoring
const healthReport = await healthChecker.runCompleteHealthCheck({
  onProgress: (message, increment) => console.log(`${message}: ${increment}%`)
});

// Privacy-focused telemetry (local only)
telemetry.trackEvent('project_analyzed', {
  framework: analysis.framework,
  endpointCount: analysis.endpoints.length,
  // No external transmission - all data stays local
});

Sample Generated Tools

// search_products tool
async function searchProducts(query: string, category?: string, priceRange?: [number, number]) {
  const params = new URLSearchParams({ q: query });
  if (category) params.append('category', category);
  if (priceRange) {
    params.append('minPrice', priceRange[0].toString());
    params.append('maxPrice', priceRange[1].toString());
  }
  
  const response = await fetch(`${API_BASE}/products/search?${params}`);
  return await response.json();
}

// create_order tool
async function createOrder(items: OrderItem[], customerInfo: CustomerInfo) {
  const response = await fetch(`${API_BASE}/orders`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ items, customer: customerInfo })
  });
  return await response.json();
}

// analyze_dataset tool
async function analyzeDataset(datasetId: string, analysisType: 'summary' | 'correlation' | 'distribution') {
  const response = await fetch(`${API_BASE}/data/${datasetId}/analyze`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ type: analysisType })
  });
  return await response.json();
}

// predict_with_model tool
async function predictWithModel(modelName: string, inputData: Record) {
  const response = await fetch(`${API_BASE}/models/${modelName}/predict`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ data: inputData })
  });
  return await response.json();
}

OpenAPI Specification

Converting an existing OpenAPI/Swagger specification into a functional MCP server.

OpenAPI 3.0 25 endpoints Auto-generated

Original OpenAPI File

api-spec/ ├── openapi.yaml ├── README.md └── examples/ ├── requests.json └── responses.json

Generation Command

agentify generate ./api-spec/openapi.yaml --type openapi --format typescript

Sample OpenAPI Spec

openapi: 3.0.0
info:
  title: Pet Store API
  version: 1.0.0
paths:
  /pets:
    get:
      summary: List all pets
      parameters:
        - name: limit
          in: query
          schema:
            type: integer
            maximum: 100
      responses:
        '200':
          description: A list of pets
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/Pet'
    post:
      summary: Create a pet
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/Pet'
      responses:
        '201':
          description: Pet created
components:
  schemas:
    Pet:
      type: object
      properties:
        id:
          type: integer
        name:
          type: string
        category:
          type: string

Generated MCP Tools

  • list_pets: Get all pets with optional filtering
  • create_pet: Add new pets to the store
  • get_pet_by_id: Retrieve specific pet information
  • update_pet: Modify existing pet details
  • delete_pet: Remove pets from the store
